def window(*args, &blk)
  win = Window.new self, extract_selector(args)

  win.use(&blk) if block_given?

  win
end

def windows(*args)
  all = @driver.window_handles.map { |handle| Window.new(self, handle: handle) }

  if args.empty?
    all
  else
    filter_windows extract_selector(args), all
  end
end
