def initialize(browser)
  if browser.is_a? Selenium::WebDriver::Driver
    Watir.logger.deprecate "Initializing `Watir::Screenshot` with a `Selenium::Driver` instance", "a `Watir::Browser` instance",
                           ids: [:screenshot_driver]
    @driver = browser
  else
    @browser = browser
    @driver = browser.wd
  end
end
